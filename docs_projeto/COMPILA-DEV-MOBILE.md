```dart
import 'package:flutter_riverpod/flutter_riverpod.dart';

final challengeControllerProvider =
    StateNotifierProvider<ChallengeController, ChallengeState>(
  (ref) => ChallengeController(ref.read),
);

class ChallengeController extends StateNotifier<ChallengeState> {
  ChallengeController(this._read) : super(ChallengeInitial());

  final Reader _read;

  Future<void> loadChallenge(String challengeId) async {
    state = ChallengeLoading();
    try {
      final challenge = await _read(challengeRepositoryProvider)
          .getChallengeById(challengeId);
      state = ChallengeLoaded(challenge);
    } catch (e) {
      state = ChallengeError(e.toString());
    }
  }

  Future<void> submitAnswer(String challengeId, String answer) async {
    state = ChallengeSubmitting();
    try {
      final result = await _read(challengeRepositoryProvider)
          .submitAnswer(challengeId, answer);

      if (result.correct) {
        state = ChallengeSuccess(result);
        // Atualizar XP localmente
        await _read(userRepositoryProvider).addXP(result.xpGained);
      } else {
        state = ChallengeFailed(result.feedback);
      }
    } catch (e) {
      state = ChallengeError(e.toString());
    }
  }
}
```
